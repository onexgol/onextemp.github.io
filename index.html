ll gcd(ll a, ll b) {
    return b == 0 ? a : gcd(b, a % b);
}

ll lcm(ll a, ll b) {
    return a / gcd(a, b) * b;
}

template <typename Container>
void print(const Container& container) {
    for (int i = 0; i < container.size(); i++) {
        if (i > 0) cout << ' ';
        cout << container[i];
    }
    cout << "\n";
}

template <typename Container>
auto summ(const Container& container) {
    return accumulate(container.begin(), container.end(), 0);
}
